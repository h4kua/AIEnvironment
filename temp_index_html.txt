        }
        if (score >= 45) {
          return {
            level: "WARNING",
            live: "WASPADA · LIVE",
            className: "warning"
          };
        }
        return {
          level: "SAFE",
          live: "AMAN · LIVE",
          className: "safe"
        };
      }

      function updateDashboardStatic() {

        riskBadge.textContent = riskMode.live;
        riskBadge.className = "risk-badge " + riskMode.className;
        riskTrend.textContent = "↑ from " + Math.round(dashboardState.lastRiskScore) + "% in last hour";
        rainfallValue.textContent = Math.round(dashboardState.rainfall);
        waterLevelValue.textContent = dashboardState.waterLevel.toFixed(2);
        bmkgValue.textContent = dashboardState.bmkgScore.toFixed(2);
        bmkgLabel.textContent = dashboardState.bmkgScore >= 0.7 ? "EXTREME" : dashboardState.bmkgScore >= 0.5 ? "ELEVATED" : "NORMAL";
        alertHeadline.textContent = "Curah hujan ekstrem — " + Math.round(dashboardState.rainfall) + " mm/h";

        regionPill.className = "region-pill " + (riskMode.className === "danger" ? "danger" : "safe");
      }

      function animateNumber(from, to, duration, formatter, update) {
        const start = performance.now();

        function step(now) {
          const progress = Math.min((now - start) / duration, 1);
          const eased = 1 - Math.pow(1 - progress, 3);
          const value = from + (to - from) * eased;
          update(formatter(value));
          if (progress < 1) {
            requestAnimationFrame(step);
          }
        }

        requestAnimationFrame(step);
      }

      function randomBetween(min, max, decimals) {
        const raw = min + Math.random() * (max - min);
        return Number(raw.toFixed(decimals || 0));
      }

      function refreshNearbyAlerts() {
        const configs = [
          { label: "Critical", score: randomBetween(88, 93, 0), distance: randomBetween(0.7, 0.9, 1) },
          { label: "High", score: randomBetween(75, 82, 0), distance: randomBetween(1.2, 1.6, 1) },
          { label: "High", score: randomBetween(71, 78, 0), distance: randomBetween(1.9, 2.3, 1) }
        ];

        nearbyAlertSubs.forEach(function (node, index) {
          const cfg = configs[index];
          node.textContent = cfg.label + " · " + cfg.score + "% · " + cfg.distance.toFixed(1) + " km";
        });
      }

      function runIngestSequence() {
        ingestLines.forEach(function (line) {
          line.classList.remove("visible");
        });

        return new Promise(function (resolve) {
          ingestLines.forEach(function (line, index) {
            setTimeout(function () {
              line.classList.add("visible");
              if (index === ingestLines.length - 1) {
                setTimeout(resolve, 360);
              }
            }, index * 400);
          });
        });
      }

      async function refreshData() {
        if (isRefreshing) {
          return;
        }

        isRefreshing = true;
        refreshButton.disabled = true;
        refreshButton.textContent = "SYNCING...";

        const previous = Object.assign({}, dashboardState);
        await runIngestSequence();

        dashboardState.lastRiskScore = previous.riskScore;
        dashboardState.riskScore = randomBetween(75, 92, 0);
        dashboardState.rainfall = randomBetween(60, 90, 0);
        dashboardState.waterLevel = randomBetween(0.78, 0.91, 2);
        dashboardState.bmkgScore = randomBetween(0.68, 0.82, 2);

        const riskMode = getRiskMode(dashboardState.riskScore);
        riskNumber.className = "risk-number " + riskMode.className;
        riskBadge.className = "risk-badge " + riskMode.className;

        animateNumber(previous.riskScore, dashboardState.riskScore, 800, function (value) {
          return Math.round(value);
        }, function (value) {
          riskNumber.textContent = value;
        });

        animateNumber(previous.rainfall, dashboardState.rainfall, 800, function (value) {
          return Math.round(value);
        }, function (value) {
          rainfallValue.textContent = value;
        });

        animateNumber(previous.waterLevel, dashboardState.waterLevel, 800, function (value) {
          return Number(value).toFixed(2);
        }, function (value) {
          waterLevelValue.textContent = value;
        });

        animateNumber(previous.bmkgScore, dashboardState.bmkgScore, 800, function (value) {
          return Number(value).toFixed(2);
        }, function (value) {
          bmkgValue.textContent = value;
        });

        setTimeout(function () {
          updateDashboardStatic();
          refreshNearbyAlerts();
          refreshButton.disabled = false;
          refreshButton.textContent = "REFRESH DATA";
          isRefreshing = false;
        }, 860);
      }

      function renderStages() {
        stageList.innerHTML = stageState.map(function (stage) {
          return (
            '<div class="stage-row ' + stage.status.toLowerCase() + '">' +
              '<span class="stage-dot"></span>' +
              '<span class="stage-name">' + stage.name + '</span>' +
              '<span class="stage-status">' + stage.status + (stage.status === "DONE" ? " ✓" : stage.status === "RUNNING" ? " ↻" : "") + '</span>' +
              '<span class="stage-time">' + (stage.timeMs ? (stage.timeMs / 1000).toFixed(2) + "s" : "0.00s") + '</span>' +
            '</div>'
          );
        }).join("");
      }

      function sleep(ms) {
        return new Promise(function (resolve) {
          setTimeout(resolve, ms);
        });
      }

      function updatePipelineButtons() {
        const label = isPipelineRunning ? '<span class="spinner"></span>PROCESSING...' : '▶ RUN PIPELINE';
        pipelineButton.innerHTML = label;
        topRunButton.innerHTML = label;
        pipelineButton.disabled = isPipelineRunning;
        topRunButton.disabled = isPipelineRunning;
      }

      async function runPipeline() {
        if (isPipelineRunning) {
          return;
        }

        isPipelineRunning = true;
        updatePipelineButtons();
        stageState = stageTemplate.map(function (stage) {
          return { name: stage.name, status: "IDLE", timeMs: 0 };
        });
        renderStages();

        for (let index = 0; index < stageState.length; index += 1) {
          const targetMs = randomBetween(80, 400, 0);
          stageState[index].status = "RUNNING";
          renderStages();

          const row = stageList.children[index];
          const timeNode = row.querySelector(".stage-time");
          const startedAt = performance.now();

          await new Promise(function (resolve) {
            function tick(now) {
              const progress = Math.min((now - startedAt) / 680, 1);
              const current = Math.round(targetMs * progress);
              timeNode.textContent = (current / 1000).toFixed(2) + "s";
              if (progress < 1) {
                requestAnimationFrame(tick);
              } else {
                resolve();
              }
            }

            requestAnimationFrame(tick);
          });

          stageState[index].status = "DONE";
          stageState[index].timeMs = targetMs;
          renderStages();
          await sleep(20);
        }

        evacuationHero.classList.add("flash");
        setTimeout(function () {
          evacuationHero.classList.remove("flash");
        }, 1400);

        lastRunValue.textContent = getNowWIB();
        isPipelineRunning = false;
        updatePipelineButtons();
      }

      function getResponse(userInput) {
        const input = userInput.toLowerCase();
        const entries = Object.entries(CHATBOT_RESPONSES);
        for (let index = 0; index < entries.length; index += 1) {
          const keyword = entries[index][0];
          const data = entries[index][1];
          if (keyword !== "default" && input.includes(keyword)) {
            return data;
          }
        }
        return CHATBOT_RESPONSES["default"];
      }

      function createMessageNode(message) {
        const messageNode = document.createElement("div");
        messageNode.className = "message " + message.role;

        const bubble = document.createElement("div");
        bubble.className = "bubble";
        const pre = document.createElement("pre");
        pre.textContent = message.text;
        bubble.appendChild(pre);
        messageNode.appendChild(bubble);

        if (message.meta) {
          const meta = document.createElement("div");
          meta.className = "message-meta";
          meta.textContent = message.meta;
          messageNode.appendChild(meta);
        }

        if (typeof message.confidence === "number") {
          const wrap = document.createElement("div");
          wrap.className = "confidence-wrap";
          const track = document.createElement("div");
          track.className = "confidence-track";
          const fill = document.createElement("div");
          fill.className = "confidence-fill";
          fill.style.width = Math.round(message.confidence * 100) + "%";
          track.appendChild(fill);
          const text = document.createElement("div");
          text.className = "confidence-text";
          text.textContent = "conf " + Math.round(message.confidence * 100) + "%";
          wrap.appendChild(track);
          wrap.appendChild(text);
          messageNode.appendChild(wrap);
        }

        return messageNode;
      }

      function appendMessage(message) {
        const node = createMessageNode(message);
        chatScroll.appendChild(node);
        chatScroll.scrollTop = chatScroll.scrollHeight;
      }

      function showTyping() {
        const typing = document.createElement("div");
        typing.className = "message typing";
        typing.id = "typingIndicator";
        typing.innerHTML = '<div class="bubble"><div class="typing-dots"><span></span><span></span><span></span></div></div>';
        chatScroll.appendChild(typing);
        chatScroll.scrollTop = chatScroll.scrollHeight;
      }

      function removeTyping() {
        const typing = document.getElementById("typingIndicator");
        if (typing) {
          typing.remove();
        }
      }

      function renderQuickReplies() {
        quickReplies.innerHTML = "";
        const activeSet = quickReplySets[Math.min(quickReplyIndex, quickReplySets.length - 1)];
        activeSet.forEach(function (chipLabel) {
          const button = document.createElement("button");
          button.className = "quick-chip";
          button.type = "button";
          button.textContent = chipLabel;
          button.addEventListener("click", function () {
            sendChatMessage(chipLabel);
            if (quickReplyIndex < quickReplySets.length - 1) {
              quickReplyIndex += 1;
              renderQuickReplies();
            }
          });
          quickReplies.appendChild(button);
        });
      }

      async function sendChatMessage(rawText) {
        const text = rawText.trim();
        if (!text) {
          return;
        }

        appendMessage({ role: "user", text: text });
        chatInput.value = "";
        chatInput.disabled = true;
        chatSend.disabled = true;
        showTyping();
        await sleep(800);
        removeTyping();

        const data = getResponse(text);
        appendMessage({
          role: "assistant",
          text: data.response,
          meta: getNowWIB().replace(" WIB", "") + " · " + data.meta,
          confidence: data.confidence
        });

        chatInput.disabled = false;
        chatSend.disabled = false;
        chatInput.focus();
      }

      function bootMessages() {
        chatScroll.innerHTML = "";
        initialMessages.forEach(function (message, index) {
          setTimeout(function () {
            appendMessage(message);
          }, index * 300);
        });
      }

      chatInput.addEventListener("keydown", function (event) {
        if (event.key === "Enter") {
